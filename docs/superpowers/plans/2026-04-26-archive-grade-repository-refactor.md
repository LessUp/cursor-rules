# Archive-Grade Repository Refactor Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Recast `cursor-rules` as an archive-grade rule library with a generated rule catalog, productized Pages/README, stable OpenSpec control docs, and lean AI/GitHub automation.

**Architecture:** Keep root-level `.mdc` files as the public contract. Add a small tested Node catalog pipeline that parses rule metadata into `docs/assets/rules.json`, then make Pages, README, and control docs converge on that single source of truth while preserving a static, low-maintenance site and only two GitHub workflows.

**Tech Stack:** Node.js (ESM, built-in `node:test`), static HTML/CSS/vanilla JS, Markdown/OpenSpec docs, GitHub Actions, GitHub CLI.

```js
import path from 'node:path';

const CATEGORY_MAP = new Map([

  ['codequality', 'general'],
  ['gitflow', 'general'],
  ['database', 'engineering'],
  ['docker', 'engineering'],
  ['python', 'language'],

  ['react', 'frontend'],
  ['vue', 'frontend'],
  ['svelte', 'frontend'],

function splitFrontmatter(content) {
  const lines = content.replace(/\r\n/g, '\n').split('\n');
  const closingIndex = lines.findIndex((line, index) => index > 0 && line.trim() === '---');
  const frontmatterLines = lines.slice(1, closingIndex);
  const bodyLines = lines.slice(closingIndex + 1);
  return { frontmatterLines, bodyLines };
}

function parseFrontmatter(lines) {
  const frontmatter = new Map();
  for (const line of lines) {
    if (!line.trim()) continue;
    const separator = line.indexOf(':');
    const key = line.slice(0, separator).trim();
    const value = line.slice(separator + 1).trim();
    frontmatter.set(key, value);
  }
  return frontmatter;
}

function parseGlobs(rawGlobs) {
  if (!rawGlobs) return [];
  const normalized = rawGlobs.replace(/^['"]|['"]$/g, '');
  if (!normalized.trim()) return [];
  return normalized.split(',').map((entry) => entry.trim()).filter(Boolean);
}

function firstHeading(lines) {
  return lines.find((line) => /^#\s+\S/.test(line))?.replace(/^#\s+/, '').trim() ?? '';
}

export async function parseRuleFile(filePath) {
  const content = await fs.readFile(filePath, 'utf8');
  const { frontmatterLines, bodyLines } = splitFrontmatter(content);
  const frontmatter = parseFrontmatter(frontmatterLines);
  const slug = path.basename(filePath, '.mdc');

  return {
    slug,
    fileName: path.basename(filePath),
    title: firstHeading(bodyLines),
    description: frontmatter.get('description')?.replace(/^['"]|['"]$/g, '') ?? '',
    globs: parseGlobs(frontmatter.get('globs') ?? ''),
    category: CATEGORY_MAP.get(slug) ?? 'other',
  };
}

export async function buildCatalog(rootDir) {
  const entries = await fs.readdir(rootDir, { withFileTypes: true });
  const files = entries.filter((entry) => entry.isFile() && entry.name.endsWith('.mdc'));
  const catalog = await Promise.all(files.map((entry) => parseRuleFile(path.join(rootDir, entry.name))));

  return catalog.sort((left, right) => left.slug.localeCompare(right.slug, 'en'));
}
```

```js
// docs/assets/site.js
const state = { language: 'zh', query: '', category: 'all', rules: [] };

async function loadRules() {
  const response = await fetch('./assets/rules.json', { cache: 'no-store' });
  if (!response.ok) throw new Error(`Failed to load rules: ${response.status}`);
  state.rules = await response.json();
}

function renderRule(rule) {
  const globs = rule.globs.length ? rule.globs.join(', ') : 'Global';
  return `
    <article class="rule-card" data-category="${rule.category}">
      <header>
        <h3>${rule.title}</h3>
        <span class="chip">${rule.category}</span>
      </header>
      <p>${rule.description}</p>
      <p class="meta"><code>${rule.fileName}</code></p>
      <p class="meta">${globs}</p>
      <a href="https://github.com/LessUp/cursor-rules/blob/master/${rule.fileName}" target="_blank" rel="noopener">Open on GitHub</a>
    </article>
  `;
}

function filteredRules() {
  return state.rules.filter((rule) => {
    const matchesCategory = state.category === 'all' || rule.category === state.category;
    const haystack = `${rule.title} ${rule.description} ${rule.fileName}`.toLowerCase();
    const matchesQuery = haystack.includes(state.query.toLowerCase());
    return matchesCategory && matchesQuery;
  });
}

function render() {
  document.querySelector('#rule-grid').innerHTML = filteredRules().map(renderRule).join('');
}

await loadRules();
render();
```
